以下是常用的100个SQL语句，涵盖数据查询、增删改、表结构操作、索引、聚合分析、多表关联等场景，适配MySQL、Oracle、SQL Server等主流数据库（部分语法有差异会标注）：


### 一、基础查询（1-15）
1. `SELECT * FROM table_name;` —— 查询表中所有列
2. `SELECT column1, column2 FROM table_name;` —— 查询指定列
3. `SELECT DISTINCT column1 FROM table_name;` —— 去重查询
4. `SELECT * FROM table_name WHERE column1 = 'value';` —— 条件查询（等于）
5. `SELECT * FROM table_name WHERE column1 > 100;` —— 条件查询（大于）
6. `SELECT * FROM table_name WHERE column1 BETWEEN 10 AND 20;` —— 范围查询
7. `SELECT * FROM table_name WHERE column1 IN ('a', 'b', 'c');` —— 枚举值查询
8. `SELECT * FROM table_name WHERE column1 LIKE '%value%';` —— 模糊查询（包含）
9. `SELECT * FROM table_name WHERE column1 LIKE 'value%';` —— 模糊查询（开头匹配）
10. `SELECT * FROM table_name WHERE column1 IS NULL;` —— 空值查询
11. `SELECT * FROM table_name WHERE column1 = 'value' AND column2 > 50;` —— 多条件且查询
12. `SELECT * FROM table_name WHERE column1 = 'value' OR column2 < 30;` —— 多条件或查询
13. `SELECT * FROM table_name ORDER BY column1 ASC;` —— 升序排序
14. `SELECT * FROM table_name ORDER BY column1 DESC;` —— 降序排序
15. `SELECT * FROM table_name LIMIT 10;` —— 限制结果行数（MySQL）


### 二、聚合与分组（16-30）
16. `SELECT COUNT(*) FROM table_name;` —— 统计总行数
17. `SELECT COUNT(column1) FROM table_name;` —— 统计非空值数量
18. `SELECT SUM(column1) FROM table_name;` —— 求和
19. `SELECT AVG(column1) FROM table_name;` —— 平均值
20. `SELECT MAX(column1) FROM table_name;` —— 最大值
21. `SELECT MIN(column1) FROM table_name;` —— 最小值
22. `SELECT column1, COUNT(*) FROM table_name GROUP BY column1;` —— 分组统计
23. `SELECT column1, SUM(column2) FROM table_name GROUP BY column1 HAVING SUM(column2) > 1000;` —— 分组后筛选
24. `SELECT column1, column2, AVG(column3) FROM table_name GROUP BY column1, column2;` —— 多列分组
25. `SELECT COUNT(DISTINCT column1) FROM table_name;` —— 去重统计
26. `SELECT CONCAT(column1, '-', column2) AS new_column FROM table_name;` —— 列拼接
27. `SELECT column1 + column2 AS total FROM table_name;` —— 列计算
28. `SELECT IF(column1 > 100, '大', '小') AS type FROM table_name;` —— 条件判断（MySQL）
29. `SELECT CASE WHEN column1 > 100 THEN '大' ELSE '小' END AS type FROM table_name;` —— 条件分支
30. `SELECT ROUND(column1, 2) FROM table_name;` —— 保留小数位数


### 三、数据插入（31-35）
31. `INSERT INTO table_name (column1, column2) VALUES ('value1', 'value2');` —— 插入指定列
32. `INSERT INTO table_name VALUES ('value1', 'value2', 'value3');` —— 插入所有列（需按表结构顺序）
33. `INSERT INTO table_name (column1, column2) SELECT columnA, columnB FROM table2;` —— 从其他表插入数据
34. `INSERT IGNORE INTO table_name (column1) VALUES ('value');` —— 忽略重复插入（MySQL）
35. `REPLACE INTO table_name (column1, column2) VALUES ('value1', 'value2');` —— 替换插入（主键冲突时更新）


### 四、数据更新（36-40）
36. `UPDATE table_name SET column1 = 'new_value' WHERE column2 = 'value';` —— 条件更新
37. `UPDATE table_name SET column1 = column1 + 10 WHERE column2 > 50;` —— 字段自增更新
38. `UPDATE table_name SET column1 = 'a', column2 = 'b' WHERE id = 1;` —— 多列更新
39. `UPDATE table1 t1 JOIN table2 t2 ON t1.id = t2.t1_id SET t1.column1 = t2.column2;` —— 多表关联更新（MySQL）
40. `UPDATE table_name SET column1 = DEFAULT WHERE id = 2;` —— 重置为默认值


### 五、数据删除（41-45）
41. `DELETE FROM table_name WHERE column1 = 'value';` —— 条件删除
42. `DELETE FROM table_name;` —— 删除所有数据（保留表结构）
43. `TRUNCATE TABLE table_name;` —— 清空表（重置自增，不可回滚）
44. `DELETE t1 FROM table1 t1 JOIN table2 t2 ON t1.id = t2.t1_id WHERE t2.column = 'value';` —— 多表关联删除（MySQL）
45. `DELETE FROM table_name ORDER BY column1 DESC LIMIT 5;` —— 删除指定行数（MySQL）


### 六、表结构操作（46-60）
46. `CREATE TABLE table_name (id INT PRIMARY KEY, name VARCHAR(50));` —— 创建表
47. `CREATE TABLE table_name LIKE old_table;` —— 复制表结构
48. `ALTER TABLE table_name ADD COLUMN age INT;` —— 添加列
49. `ALTER TABLE table_name MODIFY COLUMN age INT UNSIGNED;` —— 修改列类型
50. `ALTER TABLE table_name CHANGE COLUMN old_name new_name VARCHAR(100);` —— 修改列名和类型
51. `ALTER TABLE table_name DROP COLUMN age;` —— 删除列
52. `ALTER TABLE table_name RENAME TO new_table_name;` —— 重命名表
53. `DROP TABLE IF EXISTS table_name;` —— 删除表（存在则删）
54. `CREATE TABLE table_name (id INT AUTO_INCREMENT PRIMARY KEY);` —— 自增主键（MySQL）
55. `CREATE TABLE table_name (id INT, CONSTRAINT pk_id PRIMARY KEY (id));` —— 主键约束
56. `ALTER TABLE table_name ADD CONSTRAINT fk_t1_t2 FOREIGN KEY (t2_id) REFERENCES table2(id);` —— 外键约束
57. `ALTER TABLE table_name DROP FOREIGN KEY fk_t1_t2;` —— 删除外键
58. `CREATE TABLE table_name (name VARCHAR(50) NOT NULL);` —— 非空约束
59. `CREATE TABLE table_name (status TINYINT DEFAULT 1);` —— 默认值约束
60. `CREATE INDEX idx_column1 ON table_name(column1);` —— 创建普通索引


### 七、索引操作（61-65）
61. `CREATE UNIQUE INDEX idx_unique_column1 ON table_name(column1);` —— 唯一索引
62. `CREATE INDEX idx_composite ON table_name(column1, column2);` —— 复合索引
63. `DROP INDEX idx_column1 ON table_name;` —— 删除索引（MySQL）
64. `ALTER TABLE table_name ADD INDEX idx_column2(column2);` —— 用ALTER添加索引
65. `EXPLAIN SELECT * FROM table_name WHERE column1 = 'value';` —— 执行计划分析


### 八、多表关联查询（66-80）
66. `SELECT * FROM table1 t1 JOIN table2 t2 ON t1.id = t2.t1_id;` —— 内连接
67. `SELECT * FROM table1 t1 LEFT JOIN table2 t2 ON t1.id = t2.t1_id;` —— 左连接
68. `SELECT * FROM table1 t1 RIGHT JOIN table2 t2 ON t1.id = t2.t1_id;` —— 右连接
69. `SELECT * FROM table1 t1 FULL JOIN table2 t2 ON t1.id = t2.t1_id;` —— 全连接（Oracle/SQL Server）
70. `SELECT * FROM table1 CROSS JOIN table2;` —— 交叉连接（笛卡尔积）
71. `SELECT t1.*, t2.column FROM table1 t1 LEFT JOIN table2 t2 ON t1.id = t2.t1_id WHERE t2.id IS NULL;` —— 左连接筛选无匹配数据
72. `SELECT * FROM table1 t1 JOIN table2 t2 ON t1.id = t2.t1_id AND t1.column = 'value';` —— 连接时条件过滤
73. `SELECT t1.column1, t2.column2, t3.column3 FROM table1 t1 JOIN table2 t2 ON t1.id = t2.t1_id JOIN table3 t3 ON t2.id = t3.t2_id;` —— 多表连接
74. `SELECT * FROM (SELECT * FROM table1 WHERE column1 > 100) AS sub WHERE sub.column2 = 'a';` —— 子查询作为表
75. `SELECT * FROM table1 WHERE id IN (SELECT t1_id FROM table2 WHERE column = 'value');` —— IN子查询
76. `SELECT * FROM table1 WHERE EXISTS (SELECT 1 FROM table2 WHERE table2.t1_id = table1.id);` —— EXISTS子查询
77. `SELECT t1.id, (SELECT COUNT(*) FROM table2 WHERE t2.t1_id = t1.id) AS count FROM table1 t1;` —— 关联子查询
78. `SELECT * FROM table1 t1 JOIN (SELECT id, MAX(column1) AS max_col FROM table2 GROUP BY id) t2 ON t1.id = t2.id;` —— 子查询关联
79. `SELECT t1.column1 + t2.column2 AS total FROM table1 t1 CROSS JOIN table2 t2;` —— 列计算连接
80. `SELECT * FROM table1 NATURAL JOIN table2;` —— 自然连接（同名字段自动匹配）


### 九、高级查询（81-95）
81. `SELECT * FROM table_name WHERE column1 REGEXP '^[0-9]';` —— 正则匹配（MySQL）
82. `SELECT DATE(column_date) FROM table_name;` —— 提取日期部分
83. `SELECT YEAR(column_date) FROM table_name;` —— 提取年份
84. `SELECT DATE_ADD(column_date, INTERVAL 7 DAY) FROM table_name;` —— 日期加7天（MySQL）
85. `SELECT DATEDIFF(NOW(), column_date) FROM table_name;` —— 日期差（MySQL）
86. `SELECT * FROM table_name WHERE column_date BETWEEN '2023-01-01' AND '2023-12-31';` —— 日期范围查询
87. `SELECT * FROM table_name ORDER BY column1 LIMIT 5 OFFSET 10;` —— 分页查询（MySQL）
88. `SELECT * FROM table_name WHERE id = ?;` —— 参数化查询（防注入）
89. `SELECT column1, @row := @row + 1 AS row_num FROM table_name, (SELECT @row := 0) AS r;` —— 行号生成（MySQL）
90. `SELECT * FROM table_name WHERE column1 = LOWER('VALUE');` —— 转小写匹配
91. `SELECT * FROM table_name WHERE column1 = UPPER('value');` —— 转大写匹配
92. `SELECT SUBSTRING(column1, 1, 3) FROM table_name;` —— 字符串截取
93. `SELECT LENGTH(column1) FROM table_name;` —— 字符串长度
94. `SELECT * FROM table_name FOR UPDATE;` —— 悲观锁（事务中）
95. `SELECT column1, GROUP_CONCAT(column2 SEPARATOR ',') FROM table_name GROUP BY column1;` —— 分组拼接（MySQL）


### 十、事务与视图（96-100）
96. `START TRANSACTION;` —— 开启事务（MySQL）
97. `COMMIT;` —— 提交事务
98. `ROLLBACK;` —— 回滚事务
99. `CREATE VIEW view_name AS SELECT column1, column2 FROM table_name WHERE column3 > 50;` —— 创建视图
100. `DROP VIEW IF EXISTS view_name;` —— 删除视图


这些语句覆盖了SQL日常使用的绝大多数场景，不同数据库的语法细节（如分页、日期函数）略有差异，需结合具体数据库调整。
